        function saveIngredient() {

            const id = document.getElementById('ingredient-id').value;

            const name = document.getElementById('ingredient-name').value;

            const category = document.getElementById('ingredient-category').value;

            const unit = document.getElementById('ingredient-unit').value;

            const cost = parseFloat(document.getElementById('ingredient-cost').value);

            const stock = parseFloat(document.getElementById('ingredient-stock').value);

            const minLevel = parseFloat(document.getElementById('ingredient-min').value);

            

            if (id) {

                // Update existing ingredient

                const index = ingredients.findIndex(ing => ing.id === parseInt(id));

                if (index !== -1) {

                    ingredients[index] = { ...ingredients[index], name, category, unit, cost, stock, minLevel };

                    

                    // Log activity

                    activityLog.unshift({

                        date: new Date(),

                        item: name,

                        change: 0,

                        type: "Update",

                        user: "Current User"

                    });

                }

            } else {

                // Add new ingredient

                const newId = ingredients.length > 0 ? Math.max(...ingredients.map(ing => ing.id)) + 1 : 1;

                ingredients.push({

                    id: newId,

                    name,

                    category,

                    unit,

                    cost,

                    stock,

                    minLevel

                });

                

                // Log activity

                activityLog.unshift({

                    date: new Date(),

                    item: name,

                    change: stock,

                    type: "New Item",

                    user: "Current User"

                });

            }

            

            updateIngredientsTable();

            updateDashboard();

        }

        function updateEndMonthTable() {

            table.innerHTML = '';

            const midCount = inventoryCounts.find(count => count.type === 'mid');

            // If no mid-month count, use start count

            const prevCount = midCount || inventoryCounts.find(count => count.type === 'start');

            

            ingredients.forEach(ing => {

                const previousQty = prevCount ? 

                    (prevCount.counts.find(c => c.ingredientId === ing.id)?.quantity || ing.stock) : 

                    ing.stock;

                

                const usage = previousQty - ing.stock;

                const cost = usage * ing.cost;

                

                const row = document.createElement('tr');

                row.innerHTML = `

                    <td>${ing.name}</td>

                    <td>${ing.unit}</td>

                    <td>${previousQty}</td>

                    <td><input type="number" class="current-count" data-id="${ing.id}" value="${ing.stock}" step="0.01" min="0"></td>

                    <td>${usage.toFixed(2)}</td>

                    <td>$${cost.toFixed(2)}</td>

                `;

                table.appendChild(row);

                const endQty = endCount ? 

                    (endCount.counts.find(c => c.ingredientId === ing.id)?.quantity || midQty) : midQty;

                

                const totalUsage = startQty - endQty;

                row.innerHTML = `

                    <td>${ing.name}</td>

                    <td>${startQty.toFixed(2)}</td>
